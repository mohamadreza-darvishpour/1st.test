RESET
#how to use git reset:
#reset   --soft   or   --mixed(default)   or   --hard
# --soft   doesn't change staging index or working directory
#--mixed  is default   change staging index to match repository and doesn't change your working dirctory
#--hard   change staging index and working dir  to match repository

#--soft
git reset --soft  SHA1-CODE  # change HEAD pointer(~/Documents/gitprojects/.git/refs/heads/master) to the logged sha1-code and after SAH1-CODE won't be showed    and put the changes in staging index


#--mixed
git reset --mixed  SHA1-CODE  # change HEAD pointer(~/Documents/gitprojects/.git/refs/heads/master) to the logged sha1-code and after SAH1-CODE won't be showed      and put changes at working.dir


# --hard
git reset --hard  sha1-logged-code
# reset --hard   make sha1-logged-code equal to last logged as repo and delete all changes from working and staging dir
